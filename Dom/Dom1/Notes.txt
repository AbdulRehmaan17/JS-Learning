Selecting Elements
document.getElementById("idName");       // Select by id
document.getElementsByClassName("cls"); // Select by class (HTMLCollection)
document.getElementsByTagName("div");   // Select by tag
document.querySelector(".cls");         // First match (CSS selector)
document.querySelectorAll("p");         // All matches (NodeList)
